void code(long* v, long* k) {
  unsigned long y=v[0], z=v[1], sum=0,      /* setup */
                delta=0x9e3779b9, n=32;     /* a key schedule constant */
  while(n-- > 0) {
    y += (z<<4)+k[0] ^ z+sum ^ (z>>5)+k[1];
    z += (y<<4)+k[2] ^ z+sum ^ (y>>5)+k[3]; /* end cycle */
  }
  v[0]=y; v[1]=z;
}
